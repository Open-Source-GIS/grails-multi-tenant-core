001    package grails.plugin.multitenant.core;
002    
003    import javax.servlet.http.HttpServletRequest;
004    import java.util.Map;
005    import java.util.HashMap;
006    
007    import org.apache.log4j.Logger;
008    
009    /**
010     * Base class used for mapping sub-domains to tenantIds.  Subclasses can implement an "initialize" method
011     * that loads the data into the "hosts" map, and this class will take care of the rest.  There is also a cache
012     * that can be used to store all the domain information including the name and other information.
013     */
014    public abstract class BaseDomainNameTenantResolver implements TenantResolver
015    {
016        /**
017         * This is a logger for logging status and error messages.
018         */
019        private static Logger log = Logger.getLogger(BaseDomainNameTenantResolver.class);
020        /**
021         * This is the name of the shared cache of tenant data.  This is a shared cache where domain objects are stored
022         * with information about the existing tenants.
023         */
024        public static String TENANT_DATA_CACHE_NAME = "MultiTenantDomainData";
025        /**
026         * The map of hosts server names to tenant id used to look up tenant id.
027         */
028        protected Map<String, Integer> hosts = new HashMap<String, Integer>();
029        // This is set to true if the data has been loaded and is current
030        protected boolean loaded;
031    
032        /**
033         * This will decode the tenantId based on the server name in the request.  It will return the mapped tenant id
034         * based on the decoded server name or null if there is no match.  If there is no match the server name will be
035         * output as a fatal error.  Debugging is available to output the tenant Id and matching server name.
036         *
037         * @param inRequest - The http servlet request to decode the tenant from.
038         * @return - The tenant id based on the complete server name in the request.  Null if no match.
039         */
040        public Integer getTenantFromRequest(HttpServletRequest inRequest)
041        {
042            if (!loaded)
043            {
044                initialize();
045                loaded = true;
046            }
047    
048                    String headerName = "x-forwarded-host";
049                    String currentServerName = inRequest.getServerName();
050                    if (inRequest.getHeader(headerName) != null) {
051                            currentServerName = inRequest.getHeader(headerName);
052                            log.debug("http request URL ["+currentServerName+"] decoded using header ["+headerName+"] from request");
053                    }
054    
055            Integer tenantId = hosts.get(currentServerName);
056            if (tenantId == null)
057            {
058                log.fatal("Could not decode valid tenant id from request server " + currentServerName);
059                throw new InvalidTenantException("Could not decode mapped tenant id from request server name " + currentServerName);
060            }
061            if (log.isDebugEnabled())
062            {
063                log.debug("Decoded tenant id " + tenantId + " from http request URL " + currentServerName);
064            }
065            return tenantId;
066        }
067    
068        /**
069         * Loads up all the mappings into the protected "hosts" field.  This can do other additional operations in the
070         * super class.
071         */
072        public abstract void initialize();
073    }
