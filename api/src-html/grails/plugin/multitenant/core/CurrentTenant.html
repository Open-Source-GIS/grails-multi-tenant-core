001    package grails.plugin.multitenant.core;
002    
003    /**
004     * Interface used to store and retrieve the current tenant
005     */
006    public interface CurrentTenant {
007        public Integer get();
008    
009        public void set(Integer tenantId);
010    
011        /**
012         * The 'loaded cache' contains all the tenants that this application has encountered.
013         * Resetting the cache will erase that list, causing the app to reload data for tenants
014         * as they are re-encountered.
015         */
016        public void resetLoadedCache();
017    }
