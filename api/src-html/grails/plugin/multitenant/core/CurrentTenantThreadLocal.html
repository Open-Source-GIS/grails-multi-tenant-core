001    package grails.plugin.multitenant.core;
002    
003    import com.infusion.util.event.EventBroker;
004    import grails.plugin.multitenant.core.event.TenantChangedEvent;
005    
006    import java.util.ArrayList;
007    import java.util.List;
008    
009    /**
010     * Basic implementation that stores the current tenant in a threadlocal variable.
011     */
012    public class CurrentTenantThreadLocal implements CurrentTenant
013    {
014    // ========================================================================================================================
015    //    Static Fields
016    // ========================================================================================================================
017    
018        static ThreadLocal<Integer> currentTenant = new ThreadLocal<Integer>();
019    
020        private EventBroker eventBroker;
021        private List<Integer> loaded = new ArrayList<Integer>();
022    
023    // ========================================================================================================================
024    //    Public Instance Methods
025    // ========================================================================================================================
026    
027        public Integer get()
028        {
029            Integer rtn = currentTenant.get();
030            if (rtn == null)
031            {
032                currentTenant.set(0);
033                rtn = 0;
034            }
035            return rtn;
036        }
037    
038        public void set(Integer newTenantId)
039        {
040            Integer oldTenantId = get();
041            if (!oldTenantId.equals(newTenantId))
042            {
043                currentTenant.set(newTenantId);
044                final TenantChangedEvent changedEvent = new TenantChangedEvent(oldTenantId, newTenantId);
045                if (eventBroker != null)
046                {
047                    eventBroker.publish("tenantChanged", changedEvent);
048                    if (!loaded.contains(newTenantId))
049                    {
050                        loaded.add(newTenantId);
051                        eventBroker.publish("newTenant", changedEvent);
052                    }
053                }
054            }
055        }
056    
057        public void resetLoadedCache()
058        {
059            this.loaded.clear();
060        }
061    
062        public void setEventBroker(EventBroker eventBroker)
063        {
064            this.eventBroker = eventBroker;
065        }
066    }
