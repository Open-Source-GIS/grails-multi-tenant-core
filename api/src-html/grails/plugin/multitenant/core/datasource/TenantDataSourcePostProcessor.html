001    package grails.plugin.multitenant.core.datasource;
002    
003    import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
004    import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
005    import org.springframework.beans.factory.config.BeanDefinition;
006    import org.springframework.beans.factory.config.RuntimeBeanReference;
007    import org.springframework.beans.BeansException;
008    import org.springframework.core.Ordered;
009    import org.springframework.jdbc.datasource.TransactionAwareDataSourceProxy;
010    import org.springframework.jndi.JndiObjectFactoryBean;
011    import org.springframework.jdbc.datasource.DriverManagerDataSource;
012    import org.apache.commons.dbcp.BasicDataSource;
013    
014    /**
015     * Postprocessor that uses a tenant-aware implementation of datasource, that allows for datasource url switching.
016     */
017    public class TenantDataSourcePostProcessor implements BeanFactoryPostProcessor, Ordered {
018    // ========================================================================================================================
019    //    Public Instance Methods
020    // ========================================================================================================================
021    
022        public int getOrder() {
023            return 5;
024        }
025    
026        public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
027            final BeanDefinition beanDefinition = beanFactory.getBeanDefinition("dataSource");
028            if (JndiObjectFactoryBean.class.getName().equals(beanDefinition.getBeanClassName())) {
029                beanDefinition.setBeanClassName(TenantJndiDataSource.class.getName());
030            } else if (BasicDataSource.class.getName().equals(beanDefinition.getBeanClassName())) {
031                beanDefinition.setBeanClassName(TenantPooledDataSource.class.getName());
032            } else if (DriverManagerDataSource.class.getName().equals(beanDefinition.getBeanClassName())) {
033                beanDefinition.setBeanClassName(TenantDataSourceImpl.class.getName());
034            } else if (TransactionAwareDataSourceProxy.class.getName().equals(beanDefinition.getBeanClassName())) {
035                // Added to support Grails 1.3.X
036                beanDefinition.setBeanClassName(TenantTransactionAwareDataSourceProxy.class.getName());
037            } else {
038                return;
039            }
040            beanDefinition.setAutowireCandidate(true);
041            beanDefinition.getPropertyValues().addPropertyValue("dataSourceUrlResolver", new RuntimeBeanReference("dataSourceUrlResolver"));
042            beanDefinition.getPropertyValues().addPropertyValue("currentTenant", new RuntimeBeanReference("currentTenant"));
043        }
044    }
