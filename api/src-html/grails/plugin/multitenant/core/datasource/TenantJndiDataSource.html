001    package grails.plugin.multitenant.core.datasource;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import org.apache.log4j.Logger;
005    import org.springframework.jndi.JndiObjectFactoryBean;
006    
007    /**
008     * This class allows you to have a switching jndi datasource per tenant with the multi-tenant plugin
009     */
010    public class TenantJndiDataSource extends JndiObjectFactoryBean {
011        /**
012         * This is the resolver to be used to determine the datasource.
013         */
014        private DataSourceUrlResolver dataSourceUrlResolver;
015        /**
016         * This is the tenant this data source is associated with.
017         */
018        private CurrentTenant currentTenant;
019        /**
020         * This allows the TenantBeanFactoryPostProcessor to assign a proxy to this object in spring since there is one of
021         * there for each client.
022         */
023        public static final boolean multiTenant = true;
024        /**
025         * This is a logger for logging status or errors
026         */
027        private static Logger log = Logger.getLogger(TenantJndiDataSource.class);
028    
029        /**
030         * This method will look up the datasoure URL for the given tenant and return that url.
031         *
032         * @return The data source URL for the tenant via the resolver.
033         */
034        @Override
035        public String getJndiName() {
036            if (log.isDebugEnabled())
037                log.debug("Resolving JNDI datasource for tenant " + currentTenant.get() + " and resolver " + dataSourceUrlResolver);
038            // If we are using tenent 0 then return the default datasource
039            // TODO Can we handle a hard coded config datasource?
040            if (currentTenant.get() == 0) {
041                final String jndiNameForTenant = super.getJndiName();
042            } else {
043                // If not tenant 0 then look up the jndi name from the tenant.
044                final String jndiNameForTenant = dataSourceUrlResolver.getDataSourceUrl(currentTenant.get());
045                // If we found a jndi name we store it in the datasource otherwise we determine if it is an error or not.
046                if (jndiNameForTenant != null) {
047                    super.setJndiName(jndiNameForTenant);
048                } else {
049                    // TODO If this is single tenant we want to throw an exception to warn of a possible issue?
050                }
051            }
052            if (log.isDebugEnabled())
053                log.debug("Located JNDI datasource " + super.getJndiName() + " for tenant " + currentTenant.get());
054            return super.getJndiName();
055        }
056    
057        /**
058         * This will set the resolver type for this datasource object.
059         *
060         * @param inDataSourceUrlResolver The resolver to use to determine the datasource for the tenant.  The types could
061         *                                be database or properties file.
062         */
063    
064        public void setDataSourceUrlResolver(DataSourceUrlResolver inDataSourceUrlResolver) {
065            this.dataSourceUrlResolver = inDataSourceUrlResolver;
066        }
067    
068        /**
069         * Since this object is cached this allows you to resolve which tenant this resource applies to.
070         *
071         * @param inCurrentTenant The tenant id this data source is associated with.
072         */
073        public void setCurrentTenant(CurrentTenant inCurrentTenant) {
074            this.currentTenant = inCurrentTenant;
075        }
076    }
