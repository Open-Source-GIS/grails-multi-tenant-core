001    package grails.plugin.multitenant.core.datasource;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import org.apache.commons.dbcp.BasicDataSource;
005    
006    import java.sql.SQLException;
007    
008    import javax.sql.DataSource;
009    
010    /**
011     * This class allows pooled data sources to be used for multiple tenant databases.
012     */
013    public class TenantPooledDataSource extends BasicDataSource implements TenantDataSource {
014    // ======================================================================================================================== 
015    //    Static Fields 
016    // ======================================================================================================================== 
017    
018        public static final boolean multiTenant = true;
019    
020    // ======================================================================================================================== 
021    //    Instance Fields
022    // ======================================================================================================================== 
023    
024        private CurrentTenant currentTenant;
025        private DataSourceUrlResolver dataSourceUrlResolver;
026    
027    // ======================================================================================================================== 
028    //    Public Instance Methods
029    // ======================================================================================================================== 
030    
031        public boolean isWrapperFor(Class<?> aClass) throws SQLException {
032            throw new UnsupportedOperationException();
033        }
034    
035        public void setCurrentTenant(CurrentTenant currentTenant) {
036            this.currentTenant = currentTenant;
037        }
038    
039        public void setDataSourceUrlResolver(DataSourceUrlResolver dataSourceUrlResolver) {
040            this.dataSourceUrlResolver = dataSourceUrlResolver;
041        }
042    
043        public <T> T unwrap(Class<T> tClass) throws SQLException {
044            throw new UnsupportedOperationException();
045        }
046    
047    // ======================================================================================================================== 
048    //    Non-Public Instance Methods
049    // ======================================================================================================================== 
050    
051        @Override
052        protected DataSource createDataSource() throws SQLException {
053            final String dataSourceUrl = dataSourceUrlResolver.getDataSourceUrl(currentTenant.get());
054            if (dataSourceUrl != null) {
055                this.url = dataSourceUrl;
056            }
057            return super.createDataSource();
058        }
059    }
