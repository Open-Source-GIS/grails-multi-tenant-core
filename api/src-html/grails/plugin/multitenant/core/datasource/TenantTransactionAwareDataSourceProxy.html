001    package grails.plugin.multitenant.core.datasource;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import grails.plugin.multitenant.core.InvalidTenantException;
005    import org.apache.log4j.Logger;
006    import org.springframework.jdbc.datasource.TransactionAwareDataSourceProxy;
007    
008    import javax.naming.Context;
009    import javax.naming.InitialContext;
010    import javax.sql.DataSource;
011    
012    /**
013     * This class will intercept any calls by the grails infrastructure for a data source and return the proper
014     * data source for the current tenant.  If a transaction is already open and you make a tenant change the
015     * data source will not change.  If you want to change tenants then you need to close all open transactions and then
016     * change tenant and open the transaction.  Also you should flush and clear the hibernate session first level cache so
017     * you don't mix tenant data in the cache.  This class only supports JNDI based data sources at the current time.
018     */
019    public class TenantTransactionAwareDataSourceProxy extends TransactionAwareDataSourceProxy
020    {
021        /**
022         * This is the data source url resolver to be used to determine the data source url.
023         */
024        private DataSourceUrlResolver dataSourceUrlResolver;
025        /**
026         * This is the thread local tenant that is active at the time of the interation with the class.
027         */
028        private CurrentTenant currentTenant;
029        /**
030         * This allows the TenantBeanFactoryPostProcessor to assign a proxy to this object in spring since there is one of
031         * these for each client.
032         */
033        public static final boolean multiTenant = true;
034        /**
035         * This is a logger for logging status or errors
036         */
037        private static Logger log = Logger.getLogger(TenantTransactionAwareDataSourceProxy.class);
038    
039        /**
040         * This is the default constructor.
041         *
042         * @param targetDataSource The target data source for this class.
043         */
044        TenantTransactionAwareDataSourceProxy(DataSource targetDataSource)
045        {
046            super(targetDataSource);
047        }
048    
049        /**
050         * This method will return the actual data source for the active tenant.  If no tenant can be resolve then the
051         * default data source is returned and assumed to connect to the configuration database as tenant 0.   If the
052         * tenant is null we throw an exception.  If the tenant is 0 we return the original default which should be the
053         * default data source.  If the tenant is non zero we look up the jndi entry and return it if it is not null.  If it
054         * is null we throw an InvalidTenantException.
055         *
056         * @return The data source for the current thread local tenant.
057         */
058        @Override
059        public DataSource getTargetDataSource()
060        {
061            DataSource ds = super.getTargetDataSource();
062            Integer tenant = currentTenant.get();
063            if (tenant == null)
064            {
065                throw new InvalidTenantException("No tenant (null) returned to locate the JNDI Name for.");
066            } else if (tenant != 0)
067            {
068                // If not tenant 0 then look up the jndi name from the tenant.
069                final String jndiNameForTenant = dataSourceUrlResolver.getDataSourceUrl(tenant);
070                if (jndiNameForTenant == null)
071                {
072                    throw new InvalidTenantException("No JNDI Name returned for tenant " + tenant);
073                }
074                if (log.isDebugEnabled())
075                {
076                    log.debug("Returning the jndi dataSource " + jndiNameForTenant + " for tenant " + tenant);
077                }
078                try
079                {
080                    // TODO Support other than JNDI Data Sources
081                    // TODO Either cache the context lookup or cache the data source for performance
082                    Context ctx = new InitialContext();
083                    ds = (DataSource) ctx.lookup(jndiNameForTenant);
084                }
085                catch (Exception ex)
086                {
087                    log.fatal("Exception in Multi-tenant data source provider", ex);
088                }
089            }
090            if (log.isDebugEnabled())
091            {
092                log.debug("Returning data source for tenant 0 as" + ds);
093            }
094            return ds;
095        }
096    
097        /**
098         * This will set the resolver type for this data source object.
099         *
100         * @param inDataSourceUrlResolver The resolver to use to determine the data source for the tenant.  The types could
101         *                                be database or properties file.
102         */
103        public void setDataSourceUrlResolver(DataSourceUrlResolver inDataSourceUrlResolver)
104        {
105            this.dataSourceUrlResolver = inDataSourceUrlResolver;
106        }
107    
108        /**
109         * Since this object is cached this allows you to resolve which tenant this resource applies to.
110         *
111         * @param inCurrentTenant The tenant id this data source is associated with.
112         */
113        public void setCurrentTenant(CurrentTenant inCurrentTenant)
114        {
115            this.currentTenant = inCurrentTenant;
116        }
117    }
