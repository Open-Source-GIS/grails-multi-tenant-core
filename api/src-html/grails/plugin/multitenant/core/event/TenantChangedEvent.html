001    package grails.plugin.multitenant.core.event;
002    
003    
004    /**
005     * This class is used whenever a tenant id is changed.
006     */
007    public class TenantChangedEvent {
008        public final Integer oldTenant;
009        public final Integer newTenant;
010    
011        public TenantChangedEvent(Integer oldTenant, Integer newTenant) {
012            this.oldTenant = oldTenant;
013            this.newTenant = newTenant;
014        }
015    }
