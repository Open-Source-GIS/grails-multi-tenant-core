001    package grails.plugin.multitenant.core.groovy.compiler;
002    
003    import org.codehaus.groovy.transform.ASTTransformation;
004    import org.codehaus.groovy.transform.GroovyASTTransformation;
005    import org.codehaus.groovy.control.CompilePhase;
006    import org.codehaus.groovy.control.SourceUnit;
007    import org.codehaus.groovy.ast.ASTNode;
008    import org.codehaus.groovy.ast.ClassNode;
009    import org.codehaus.groovy.ast.expr.ConstantExpression;
010    import org.codehaus.groovy.grails.compiler.injection.GrailsASTUtils;
011    import org.apache.commons.logging.LogFactory;
012    import org.apache.commons.logging.Log;
013    
014    import java.lang.reflect.Modifier;
015    
016    /**
017     * Performs an ast transformation on a class - adds a tenantId property to the subject class.
018     */
019    @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
020    public class TenantASTTransformation implements ASTTransformation {
021    // ========================================================================================================================
022    //    Static Fields
023    // ========================================================================================================================
024    
025        private static final Log LOG = LogFactory.getLog(TenantASTTransformation.class);
026    
027    // ========================================================================================================================
028    //    Public Instance Methods
029    // ========================================================================================================================
030    
031        public void visit(ASTNode[] astNodes, SourceUnit sourceUnit) {
032            for (ASTNode astNode : astNodes) {
033                if (astNode instanceof ClassNode) {
034                    ClassNode classNode = (ClassNode) astNode;
035                    final boolean hasTenantId = GrailsASTUtils.hasOrInheritsProperty(classNode, "tenantId");
036                    if (!hasTenantId) {
037                        LOG.info("[TenantASTTransformation] Adding property [tenantId] to class [" + classNode.getName() + "]");
038                        classNode.addProperty("tenantId", Modifier.PUBLIC, new ClassNode(Integer.class), new ConstantExpression(0), null, null);
039                    }
040                }
041            }
042        }
043    }
