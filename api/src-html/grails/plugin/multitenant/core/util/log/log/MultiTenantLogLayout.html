001    package grails.plugin.multitenant.core.util.log.log;
002    
003    import org.apache.log4j.PatternLayout;
004    import org.apache.log4j.helpers.PatternParser;
005    
006    /**
007     * This class will set up a pattern to support inserting the tenant id or name into the beginning of any logging
008     * pattern. It will either enhance an existing pattern or create a new pattern with the tenant id or tenant name
009     * at the beginning of the patter.
010     */
011    public class MultiTenantLogLayout extends PatternLayout
012    {
013        /**
014         * Inserts a %T after the $t (thread logging) to display the current tenant mapped id.
015         */
016        public static final String DEFAULT_TENANT_CONVERSION_PATTERN = "%d [%t] [tenant %T] %-5p %c{2} %x - %m%n";
017    
018        /**
019         * This is a constructor that does not take a pattern and there fore uses the defined default pattern.
020         */
021        public MultiTenantLogLayout()
022        {
023            this(DEFAULT_TENANT_CONVERSION_PATTERN);
024        }
025    
026        /**
027         * This constructor allows for creation using a user defined pattern
028         *
029         * @param inPattern - The user defined logging pattern that can contain a %T where the tenant id will be inserted.
030         */
031        public MultiTenantLogLayout(String inPattern)
032        {
033            super(inPattern);
034        }
035    
036        /**
037         * This will create a pattern parser using the defined user pattern or the default pattern if one is not supplied.
038         * This parser knows how to insert the current tenant id in to the proper location in the pattern.
039         *
040         * @param inPattern - The logging pattern to use during the parsing operation.
041         * @return
042         */
043        public PatternParser createPatternParser(String inPattern)
044        {
045            return new MultiTenantPatternParser(
046                    inPattern == null ? DEFAULT_TENANT_CONVERSION_PATTERN : inPattern);
047        }
048    }
